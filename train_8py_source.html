153 def init_env(
154  max_episode_duration: float,
155  show: bool,
156  spine_path: str,
157 ):
158  """!
159  Get an environment initialization function for a set of parameters.
160 
161  @param max_episode_duration Maximum duration of an episode, in seconds.
162  @param show If true, show simulator GUI.
163  @param spine_path Path to the Bullet spine binary.
164  """
165  env_settings = EnvSettings()

167 
168  def _init():
169  shm_name = f"/{get_random_word()}"
170  pid = os.fork()
171  if pid == 0: # child process: spine
172  argv = get_bullet_argv(shm_name, show=show)
173  os.execvp(spine_path, ["bullet"] + argv)
174  return
175 
176  # parent process: trainer
177  agent_frequency = env_settings.agent_frequency
178  velocity_env = gymnasium.make(
179  env_settings.env_id,
180  max_episode_steps=int(max_episode_duration * agent_frequency),
181  frequency=agent_frequency,
182  regulate_frequency=False,
183  reward_weights=upkie.envs.UpkieGroundVelocity.RewardWeights(
184  **env_settings.reward_weights
185  ),
186  shm_name=shm_name,
187  spine_config=env_settings.spine_config,
188  max_ground_velocity=env_settings.max_ground_velocity,
189  )
190  velocity_env.reset(seed=seed)
191  velocity_env._prepatch_close = velocity_env.close
192 
193  def close_monkeypatch():
194  logging.info(f"Terminating spine {shm_name} with {pid=}...")
195  os.kill(pid, signal.SIGINT) # interrupt spine child process
196  os.waitpid(pid, 0) # wait for spine to terminate
197  velocity_env._prepatch_close()
198 
199  velocity_env.close = close_monkeypatch
200  env = make_ppo_balancer_env(velocity_env, env_settings, training=True)
201  return Monitor(env)
202 
203  set_random_seed(seed)
204  return _init

217 def affine_schedule(y_0: float, y_1: float) -> Callable[[float], float]:
218  """!
219  Affine schedule as a function over the [0, 1] interval.
220 
221  @param y_0 Function value at zero.
222  @param y_1 Function value at one.
223  @return Corresponding affine function.
224  """
225  diff = y_1 - y_0

228  """!
229  Compute the current learning rate from remaining progress.
230 
231  @param x Progress decreasing from 1 (beginning) to 0.
232  @return Corresponding learning rate>

234  return y_0 + x * diff
