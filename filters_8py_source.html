39 def bounded_derivative_filter(
40  prev_output: float,
41  new_input: float,
42  dt: float,
43  output_bounds: Tuple[float, float],
44  derivative_bounds: Tuple[float, float],
45 ) -> float:
46  """!
47  Filter signal so that its output and output derivative stay within bounds.
48 
49  @param prev_output Previous filter output, or initial value.
50  @param new_input New filter input.
51  @param dt Sampling period in [s].
52  @param output_bounds Min and max value for the output.
53  @param derivative_bounds Min and max value for the output derivative.
54  @returns New filter output.
55  """
56  derivative = (new_input - prev_output) / dt
57  derivative = clamp(derivative, *derivative_bounds)
58  output = prev_output + derivative * dt
59  return clamp(output, *output_bounds)
