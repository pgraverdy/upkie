34 void FloorContact::read(const Dictionary& observation) {
35  if (params_.incomplete()) {
36  spdlog::warn("[FloorContact] Observer is not configured");
37  return;
38  }
39  if (!observation.has("servo")) {
40  return;
41  }
42  const bool at_least_one_wheel_contact = check_wheel_contacts(observation);
43  update_upper_leg_torque(observation);
44  contact_ = (at_least_one_wheel_contact ||
45  (upper_leg_torque_ > params_.upper_leg_torque_threshold));
46 }

48 bool FloorContact::check_wheel_contacts(const Dictionary& observation) {
49  const auto& servo = observation("servo");
50  bool at_least_one_contact = false;
51  for (const auto& wheel : params_.wheels) {
52  auto it = wheel_contacts_.find(wheel); // found by construction
53  auto& wheel_contact = it->second;
54  try {
55  double velocity = servo(wheel)("velocity");
56  double torque = servo(wheel)("torque");
57  wheel_contact.observe(torque, velocity, params_.dt);
58  if (observation.has("joystick") &&
59  observation("joystick").get<bool>("cross_button")) {
60  wheel_contact.reset_contact();
61  } else if (wheel_contact.contact()) {
62  at_least_one_contact = true;
63  }
64  } catch (const KeyError& e) {
65  spdlog::warn("[FloorContact] No observation for {} yet", wheel);
66  continue;
67  }
68  }
69  return at_least_one_contact;
70 }

72 void FloorContact::update_upper_leg_torque(const Dictionary& observation) {
73  const auto& servo = observation("servo");
74  double squared_torques = 0.0;
75  for (const auto& joint : params_.upper_leg_joints) {
76  try {
77  const double torque = servo(joint)("torque");
78  squared_torques += torque * torque;
79  } catch (const KeyError& e) {
80  spdlog::warn("[FloorContact] No observation for {} yet", joint);
81  continue;
82  }
83  }
84  constexpr double kTorqueCutoffPeriod = 0.01; // [s]
85  const double new_torque = std::sqrt(squared_torques);
86  upper_leg_torque_ = low_pass_filter(upper_leg_torque_, kTorqueCutoffPeriod,
87  new_torque, params_.dt);
88 }
