16 def compute_pitch_frame_in_parent(
17  orientation_frame_in_parent: NDArray[float],
18 ) -> float:
19  """!
20  Get pitch angle of a given frame relative to the parent vertical.
21 
22  Figure:
23 
24  ```
25  parent z
26  ^ frame z
27  | /
28  | /
29  | /
30  | /
31  | /
32  |/
33  (x)-----------> heading vector
34  \\ )
35  \\+ positive pitch
36  \\
37  \\
38  \\
39  frame x
40  ```
41 
42  @param orientation_frame_in_parent Rotation matrix from the target frame to
43  the parent frame.
44 
45  @returns Angle from the parent z-axis (gravity) to the frame z-axis.
46  Equivalently, angle from the heading vector to the sagittal vector of
47  the frame.
48 
49  @note Angle is positive in the trigonometric sense (CCW positive, CW
50  negative) in the heading-vertical plane directed by the lateral vector,
51  which is ``(x)`` in the above schematic (pointing away) and not ``(.)``
52  (poiting from screen to the reader).
53  """
54  sagittal = orientation_frame_in_parent[:, 0]
55  sagittal /= np.linalg.norm(sagittal) # needed for prec around cos_pitch=0.
56  heading_in_parent = sagittal - sagittal[2] * np.array([0.0, 0.0, 1.0])
57  heading_in_parent /= np.linalg.norm(heading_in_parent)
58  if orientation_frame_in_parent[2, 2] < 0:
59  heading_in_parent *= -1.0
60  sign = +1 if sagittal[2] < 0 else -1
61  cos_pitch = clamp(np.dot(sagittal, heading_in_parent), -1.0, 1.0)
62  pitch = float(sign * np.arccos(cos_pitch)) # float rather than np.float64
63  return pitch

66 def compute_base_orientation_from_imu(
67  quat_imu_in_ars: Tuple[float, float, float, float],
68  rotation_base_to_imu: Optional[NDArray[float]] = None,
69 ) -> NDArray[float]:
70  """!
71  Get the orientation of the base frame with respect to the world frame.
72 
73  @param quat_imu_in_ars Quaternion representing the rotation matrix from the
74  IMU frame to the attitude reference system (ARS) frame, in ``[w, x, y,
75  z]`` format.
76  @param rotation_base_to_imu Rotation matrix from the base frame to the IMU
77  frame. When not specified, the default Upkie mounting orientation is
78  used.
79  @returns Rotation matrix from the base frame to the world frame.
80  """
81  if rotation_base_to_imu is None:
82  # Default Upkie mounting orientation: USB connectors of the raspi
83  # pointing to the left of the robot (XT-30 of the pi3hat to the right)
84  rotation_base_to_imu = np.diag([-1.0, 1.0, -1.0])
85 
86  rotation_imu_to_ars = rotation_matrix_from_quaternion(quat_imu_in_ars)
87 
88  # The attitude reference system frame has +x forward, +y right and +z down,
89  # whereas our world frame has +x forward, +y left and +z up:
90  # https://github.com/mjbots/pi3hat/blob/ab632c82bd501b9fcb6f8200df0551989292b7a1/docs/reference.md#orientation
91  rotation_ars_to_world = np.diag([1.0, -1.0, -1.0])
92 
93  rotation_base_to_world = (
94  rotation_ars_to_world @ rotation_imu_to_ars @ rotation_base_to_imu
95  )
96  return rotation_base_to_world

122 def compute_base_angular_velocity_from_imu(
123  angular_velocity_imu_in_imu: NDArray[float],
124 ) -> NDArray[float]:
125  r"""!
126  Compute the body angular velocity of the base from IMU readings.
127 
128  @param angular_velocity_imu_in_imu Angular velocity from the IMU.
129  @returns Body angular velocity of the base frame.
130 
131  Calculation checks:
132 
133  - \f$I\f$: IMU frame
134  - \f$B\f$: base frame
135  - \f$W\f$: world (inertial) frame
136 
137  Our input is \f${}_I \omega_{WI}\f$, we seek \f${}_B \omega_{WB}\f$.
138 
139  \f$
140  \begin{align*}
141  \dot{R}_{WI} & = R_{WI} ({}_I \omega_{WI} \times) \\
142  R_{WB} & = R_{WI} R_{IB} \\
143  \dot{R}_{WB} & = \dot{R}_{WI} R_{IB} \\
144  & = R_{WI} ({}_I \omega_{WI} \times) R_{IB} \\
145  & = R_{WI} R_{IB} ({}_B \omega_{WI} \times) \\
146  & = R_{WB} ({}_B \omega_{WI} \times) = R_{WB} ({}_B \omega_{WB} \times)
147  \end{align*}
148  \f$
149 
150  Thus \f${}_B \omega_{WB} = R_{BI} {}_I \omega_{WI}\f$.
151  """
152  # TODO(scaron): move to config
153  rotation_base_to_imu = np.diag([-1.0, 1.0, -1.0])
154  rotation_imu_to_base = rotation_base_to_imu.T
155  angular_velocity_base_in_base = (
156  rotation_imu_to_base @ angular_velocity_imu_in_imu
157  )
158  return angular_velocity_base_in_base
