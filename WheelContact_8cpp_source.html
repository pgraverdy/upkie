20 void WheelContact::observe(const double torque, const double velocity,
21  const double dt) noexcept {
22  if (params_.cutoff_period < 1e-6) {
23  spdlog::warn("[WheelContact] Observer is not configured");
24  return;
25  }
26 
27  const double prev_velocity = velocity_;
28  velocity_ = low_pass_filter(velocity_, params_.cutoff_period, velocity, dt);
29  const double new_acceleration = (velocity_ - prev_velocity) / dt;
30  abs_acceleration_ = low_pass_filter(abs_acceleration_, params_.cutoff_period,
31  std::abs(new_acceleration), dt);
32  abs_torque_ =
33  low_pass_filter(abs_torque_, params_.cutoff_period, std::abs(torque), dt);
34 
35  // Don't detect touchdown when averaged absolute acceleration or torque are
36  // too low. Typically when the robot is in the air torques are low. This
37  // avoids false positives when the air wheel acceleration happens to be low.
38  if (!contact_ && (abs_acceleration_ < params_.min_touchdown_acceleration ||
39  abs_torque_ < params_.min_touchdown_torque)) {
40  return;
41  }
42 
43  inertia_ = abs_torque_ / (abs_acceleration_ + 1e-4);
44  if (inertia_ < params_.liftoff_inertia) {
45  contact_ = false;
46  } else if (inertia_ > params_.touchdown_inertia) {
47  contact_ = true;
48  }
49 }
