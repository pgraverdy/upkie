49 def get_tip_state(
50  observation, tip_height: float = 0.58
51 ) -> Tuple[float, float]:
52  """!
53  Compute the state of the virtual tip used in the agent's reward.
54 
55  This extra info is for logging only.
56 
57  @param observation Observation vector.
58  @param tip_height Height of the virtual tip.
59  @returns Pair of tip (position, velocity) in the sagittal plane.
60  """
61  pitch = observation[0]
62  ground_position = observation[1]
63  angular_velocity = observation[2]
64  ground_velocity = observation[3]
65  tip_position = ground_position + tip_height * np.sin(pitch)
66  tip_velocity = ground_velocity + tip_height * angular_velocity * np.cos(
67  pitch
68  )
69  return tip_position, tip_velocity
