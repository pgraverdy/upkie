27 double WheelOdometry::compute_average_velocity(const Dictionary& floor_contact,
28  const Dictionary& servo) {
29  if (params_.signed_radius.empty()) {
30  throw std::runtime_error(
31  "[WheelOdometry] Observer not configured: 'signed_radius' is empty");
32  }
33 
34  double velocity_sum = 0.0;
35  unsigned nb_wheels_in_contact = 0u;
36  for (const auto& wheel_radius_pair : params_.signed_radius) {
37  const auto& wheel = wheel_radius_pair.first;
38  const bool wheel_contact = floor_contact(wheel).get<bool>("contact");
39  if (!wheel_contact) {
40  continue;
41  }
42  const double wheel_velocity = servo(wheel)("velocity"); // [rad] / [s]
43  const double signed_radius = wheel_radius_pair.second;
44  const double linear_velocity = signed_radius * wheel_velocity; // [m] / [s]
45  velocity_sum += linear_velocity;
46  ++nb_wheels_in_contact;
47  }
48 
49  if (nb_wheels_in_contact == 0u) {
50  spdlog::warn("[WheelOdometry] Contact detected, but no wheel in contact?");
51  return 0.0;
52  }
53 
54  return (velocity_sum / nb_wheels_in_contact);
55 }
