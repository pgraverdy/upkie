123 
124  @param env Gym environment to Upkie.
125  @param nb_env_steps Number of environment steps to perform (zero to run

127  @param rebuild_qp_every_time If set, rebuild all QP matrices at every
128  iteration. Otherwise, only update vectors.
129  @param show_live_plot Show a live plot.
130  @param stage_input_cost_weight Weight for the stage input cost.
131  @param stage_state_cost_weight Weight for the stage state cost.
132  @param terminal_cost_weight Weight for the terminal cost.
133  @param warm_start If set, use the warm-starting feature of ProxQP.
134  """
135  pendulum = PendularUpkie()
136  mpc_problem = pendulum.build_mpc_problem(
137  terminal_cost_weight=terminal_cost_weight,
138  stage_state_cost_weight=stage_state_cost_weight,
139  stage_input_cost_weight=stage_input_cost_weight,
140  )
141  mpc_problem.initial_state = np.zeros(4)
142  mpc_qp = MPCQP(mpc_problem)
143  proxqp = ProxQPWorkspace(mpc_qp)
144 
145  live_plot = None
146  if show_live_plot and not on_raspi():
147  from qpmpc.live_plots import ( # imports matplotlib
148  WheeledInvertedPendulumPlot,
149  )
150 
151  live_plot = WheeledInvertedPendulumPlot(pendulum, order="velocities")
152 
153  env.reset() # connects to the spine
154  commanded_velocity = 0.0
155  action = np.zeros(env.action_space.shape)
156 
157  planning_times = np.empty((nb_env_steps,)) if nb_env_steps > 0 else None
158  base_pitches = np.empty((nb_env_steps,)) if nb_env_steps > 0 else None
159  step = 0
160  while True:
161  action[0] = commanded_velocity
162  observation, _, terminated, truncated, info = env.step(action)
163  if terminated or truncated:
164  observation, info = env.reset()
165  commanded_velocity = 0.0
166 
167  spine_observation = info["spine_observation"]
168  floor_contact = spine_observation["floor_contact"]["contact"]
169 
170  # Unpack observation into initial MPC state
171  (
172  base_pitch,
173  ground_position,
174  base_angular_velocity,
175  ground_velocity,
176  ) = observation
177  initial_state = np.array(
178  [
179  ground_position,
180  base_pitch,
181  ground_velocity,
182  base_angular_velocity,
183  ]
184  )
185 
186  nx = WheeledInvertedPendulum.STATE_DIM
187  target_states = np.zeros((pendulum.nb_timesteps + 1) * nx)
188  mpc_problem.update_initial_state(initial_state)
189  mpc_problem.update_goal_state(target_states[-nx:])
190  mpc_problem.update_target_states(target_states[:-nx])
191 
192  t0 = perf_counter()
193  if rebuild_qp_every_time:
194  plan = solve_mpc(mpc_problem, solver="proxqp")
195  else:
196  mpc_qp.update_cost_vector(mpc_problem)
197  if warm_start:

199  else:
200  qpsol = solve_problem(mpc_qp.problem, solver="proxqp")
201  plan = Plan(mpc_problem, qpsol)
202  if nb_env_steps > 0:
203  base_pitches[step] = base_pitch
204  planning_times[step] = perf_counter() - t0
205 
206  if not floor_contact:
207  commanded_velocity = low_pass_filter(
208  prev_output=commanded_velocity,
209  cutoff_period=0.1,
210  new_input=0.0,
211  dt=env.dt,
212  )
213  elif plan.is_empty:
214  logging.error("Solver found no solution to the MPC problem")
215  logging.info("Continuing with previous action")
216  else: # plan was found
217  pendulum.state = initial_state
218  if live_plot is not None:
219  t = time.time()
220  live_plot.update(plan, t, initial_state, t)
221  commanded_accel = plan.first_input
222  commanded_velocity = clamp_and_warn(
223  commanded_velocity + commanded_accel * env.dt / 2.0,
224  lower=-1.0,
225  upper=+1.0,
226  label="commanded_velocity",
227  )
228 
229  if nb_env_steps > 0:
230  step += 1
231  if step >= nb_env_steps:
232  break
233 
234  report(mpc_problem, mpc_qp, planning_times)
235  np.save("base_pitches.npy", base_pitches)
236  np.save("planning_times.npy", planning_times)
