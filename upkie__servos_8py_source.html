28  """!
29  Upkie with full observation and joint position-velocity-torque actions.
30 
31  ### Action space
32 
33  The action space is a dictionary with one key for each servo on Upkie:
34 
35  - ``left_hip``: Left hip joint (qdd100)
36  - ``left_hip``: Left knee joint (qdd100)
37  - ``left_hip``: Left wheel joint (mj5208)
38  - ``right_hip``: Right hip joint (qdd100)
39  - ``right_hip``: Right knee joint (qdd100)
40  - ``right_hip``: Right wheel joint (mj5208)
41 
42  The value for each dictionary is itself a dictionary for each
43 
44  - ``position``: Joint angle in [rad] (NaN to disable) (required).
45  - ``velocity``: Joint velocity in [rad] / [s] (required).
46  - ``feedforward_torque``: Joint torque in [N] * [m].
47  - ``kp_scale``: Scaling factor applied to the position feedback gain,
48  between zero and one.
49  - ``kd_scale``: Scaling factor applied to the velocity feedback gain,
50  between zero and one.
51  - ``maximum_torque``: Maximum joint torque (feedforward + feedback) during
52  the whole actuation step, in [N] * [m].
53 
54  ### Observation space
55 
56  Obversations from this environment are full dictionary reported by the
57  spine. See @ref observations.
58 
59  ### Attributes
60 
61  The environment has the following attributes:
62 
63  - ``robot``: Pinocchio robot wrapper.
64  - ``version``: Environment version number.
65  """

333  def get_spine_action(self, env_action: dict) -> dict:
334  """!
335  Convert environment action to a spine action dictionary.
336 
337  @param env_action Environment action.
338  @returns Spine action dictionary.
339  """
340  spine_action = {"servo": {}}
341  for joint in self.JOINT_NAMES:
342  servo_action = {}
343  for key in self.ACTION_KEYS:
344  try:
345  action = (
346  env_action[joint][key]
347  if key in env_action[joint]
348  else self.__default_action__default_action[joint][key]
349  )
350  except KeyError as key_error:
351  raise ActionError(
352  f'Missing key "{key}" required for joint "{joint}"'
353  ) from key_error
354  servo_action[key] = clamp_and_warn(
355  action,
356  self.__min_action__min_action[joint][key],
357  self.__max_action__max_action[joint][key],
358  label=f"{joint}: {key}",
359  )
360  spine_action["servo"][joint] = servo_action
361  return spine_action
